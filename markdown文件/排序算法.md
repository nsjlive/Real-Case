### 排序算法
//选择排序与冒泡排序的区别：
冒泡排序交换的次数会多一些，最坏情况是n^2/2次；选择排序最坏的情况是n次。

1. 选择排序

	```java
	public void selectSort(int [] arr){
		for(int i=0;i<arr.length;i++){
			int k=i;
			for(int j=i+1;j<arr.length;j++){//找到最小值对应的索引为k
				if(arr[j]<arr[k]){
					k=j;
				}
			}
			if(k!=i){//k若为当前i,则无需交换。交换的目的是让最小的元素下标为当前的i。
			int temp=arr[i];
			arr[i]=arr[k];
			arr[k]=temp;
			}
		}
	}
	```

	2.插入排序

	```java
	public void insertSort(int[] arr){
		int len=arr.length;
		for(int i=1;i<len;i++){
			int cur=arr[i];//用做比较的数据
			int preIndex=i-1;
			while(preIndex>=0&&arr[preIndex]>cur){
				arr[preIndex+1]=arr[preIndex];
				preIndex--;
			}
			arr[preIndex+1]=cur;//假设只有两个元素，实际上相当于交换
		}
	}
	```

	3. 冒泡排序 

	```java
	public void bubbleSort(int [] arr){
		int len=arr.length;
		for(int i=0;i<len-1;i++){
			for(j=0;j<len-1-i;j++){//注意：不要写成j=i,冒泡排序j每次是从下标为0的元素开始
				if(arr[j] >arr[j+1]){
				int temp=arr[j];
				arr[j]=arr[j+1];
				arr[j+1]=temp;
				}
			}
		}
	}
	```

	4. 快速排序
	
	* 填坑法 参考：[this](<https://segmentfault.com/a/1190000004410119>)
	
	```java
	 public  static void quickSort(int [] arr,int low,int high){
	        if (low < high) {
	            /**
	             * /找寻基准数据的正确索引index
	             * 保证左侧元素比arr[index]小，右边比它大
	             */
	            int index=getIndex(arr,low,high);
	
	            // 对index元素左右进行迭代
	            quickSort(arr,low,index-1);
	            quickSort(arr,index+1,high);
	        }
	    }
	
	    public static int getIndex(int [] arr,int low ,int high){
	        int temp=arr[low];
	        while(low<high){
	            while(low<high&&arr[high]>=temp){
	                high--;
	            }
	            arr[low]=arr[high];//小的值放前面
	            while(low<high&&arr[low]<=temp){
	                low++;
	            }
	            arr[high]=arr[low];//大的值放后面
	        }
	        arr[low]=temp;//填坑
	        return low;
	    }
	```
	
	
归并排序：
```java
public static void mergeSort(int[] arr) {
    sort(arr, 0, arr.length - 1);
}

public static void sort(int[] arr, int L, int R) {
    if(L == R) {
        return;
    }
    int mid = L + ((R - L) >> 1);
    sort(arr, L, mid);
    sort(arr, mid + 1, R);
    merge(arr, L, mid, R);
}

public static void merge(int[] arr, int L, int mid, int R) {
    int[] temp = new int[R - L + 1];
    int i = 0;
    int p1 = L;
    int p2 = mid + 1;
    // 比较左右两部分的元素，哪个小，把那个元素填入temp中
    while(p1 <= mid && p2 <= R) {
        temp[i++] = arr[p1] < arr[p2] ? arr[p1++] : arr[p2++];
    }
    // 上面的循环退出后，把剩余的元素依次填入到temp中
    // 以下两个while只有一个会执行
    while(p1 <= mid) {
        temp[i++] = arr[p1++];
    }
    while(p2 <= R) {
        temp[i++] = arr[p2++];
    }
    // 把最终的排序的结果复制给原数组
    for(i = 0; i < temp.length; i++) {
        arr[L + i] = temp[i];
    }
}
```
